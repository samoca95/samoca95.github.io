# Classes and inheritance
class Animal:
  def __init__(self, name):
      self.name = name
  
  def speak(self):
      return f"{self.name} makes a sound."

class Dog(Animal):
  def speak(self):
      return f"{self.name} barks."

dog = Dog("Rex")
print(dog.speak())

# Decorators
def decorator(func):
  def wrapper():
      print("Before function call")
      func()
      print("After function call")
  return wrapper

@decorator
def say_hello():
  print("Hello!")

say_hello()
